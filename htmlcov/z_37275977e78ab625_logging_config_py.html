174class InterceptHandler(logging.Handler): 

175 """Intercept standard logging and forward to loguru. 

176 

177 This allows capturing logs from dependencies (litellm, instructor, etc.) 

178 into our unified logging system. 

179 

180 Instead of trying to trace call frames (which is fragile), 

181 we use the record's built-in location info. 

182 """

184 def emit(self, record: logging.LogRecord) -> None: 

185 # Get corresponding loguru level 

186 try: 

187 level = logger.level(record.levelname).name 

188 except ValueError: 

189 level = record.levelno 

190 

191 # Use record's built-in location info instead of frame tracing 

192 # This is more reliable for intercepted logs 

193 logger.bind( 

194 name=record.name, 

195 module=record.module, 

196 function=record.funcName, 

197 line=record.lineno, 

198 ).opt(exception=record.exc_info).log(level, record.getMessage())

201def _format_json_log(record: Any) -> str: 

202 """Loguru format callable producing one JSON object per line. 

203 

204 json.dumps handles escaping (quotes, backslashes, newlines) in the 

205 message. The serialized payload is stashed in record["extra"] because 

206 the returned string is still treated as a loguru format template. 

207 """ 

208 record["extra"]["_json"] = json.dumps( 

209 { 

210 "ts": record["time"].strftime("%Y-%m-%dT%H:%M:%S"), 

211 "lvl": record["level"].name, 

212 "src": f"{record['module']}:{record['line']}", 

213 "msg": record["message"], 

214 }, 

215 ensure_ascii=False, 

216 ) 

217 return "{extra[_json]}\n"

345def _setup_log_interception() -> None: 

346 """Intercept third-party library logs and route to loguru. 

347 

348 Intercepted loggers only capture WARNING+ to reduce noise. 

349 """ 

350 intercept_handler = InterceptHandler() 

351 

352 for logger_name in INTERCEPTED_LOGGERS: 

353 stdlib_logger = logging.getLogger(logger_name) 

354 stdlib_logger.handlers.clear() # Remove existing handlers 

355 stdlib_logger.addHandler(intercept_handler) 

356 stdlib_logger.propagate = False # Don't propagate to root logger 

357 # Only capture WARNING+ from third-party libs to reduce noise 

358 stdlib_logger.setLevel(logging.WARNING)

479def _should_show_log(record: Any, verbose: bool) -> bool: 

480 """Filter function for console logging. 

481 

482 Console only shows INFO+ (DEBUG goes to file only). 

483 verbose mode shows more detail in INFO messages, not DEBUG. 

484 

485 Args: 

486 record: Loguru Record object 

487 verbose: Whether verbose mode is enabled 

488 

489 Returns: 

490 True if the log should be shown 

491 """ 

492 level = record["level"].name 

493 

494 # DEBUG never goes to console (file only) 

495 if level == "DEBUG": 495 ↛ 496line 495 didn't jump to line 496 because the condition on line 495 was never true

496 return False 

497 

498 # Get the module/name from the record (bound by InterceptHandler for stdlib logs, 

499 # or from native loguru record fields for first-party logs). 

500 name = _get_record_field(record, "name") 

501 module = _get_record_field(record, "module") 

502 

503 # Always show warnings and above, except known third-party retry-loop 

504 # noise that duplicates markitai's own failure summaries (see 

505 # _is_third_party_retry_noise). 

506 if level in ("WARNING", "ERROR", "CRITICAL"): 

507 return not _is_third_party_retry_noise(record, name) 

508 

509 # Filter out third-party INFO 

510 if level == "INFO" and _is_third_party_log(name, module): 510 ↛ 511line 510 didn't jump to line 511 because the condition on line 510 was never true

511 return False 

512 

513 # In non-verbose mode, filter out ALL INFO logs from console 

514 # User-facing output uses ui.* functions (ui.success, ui.summary, etc.) 

515 # which write directly to console, not through logger 

516 if not verbose and level == "INFO": 

517 return False 

518 

519 if level == "INFO" and _is_internal_info_noise(record): 

520 return False 

521 

522 return True
