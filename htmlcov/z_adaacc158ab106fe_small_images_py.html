13def remove_small_images(root: Tag, min_size: int = _MIN_SIZE) -> int: 

14 """Remove images and SVGs smaller than *min_size* pixels.

16 Only removes when BOTH dimensions are known and below the threshold.

18 

19 Args: 

20 root: Content root element. 

21 min_size: Minimum dimension in pixels. 

22 

23 Returns: 

24 Number of elements removed. 

25 """ 

26 removed = 0

28 w, h = _get_dimensions(el)

35def _get_dimensions(el: Tag) -> tuple[int | None, int | None]: 

36 """Extract width and height from attributes or inline style.""" 

37 w = _parse_int(el.get("width")) 

38 h = _parse_int(el.get("height")) 

39 if w is not None and h is not None: 

40 return w, h 

41 

42 style = el.get("style", "") 

43 if style: 

44 for prop, val in _DIMENSION_STYLE_RE.findall(str(style)): 

45 if prop == "width" and w is None: 

46 w = int(val)
