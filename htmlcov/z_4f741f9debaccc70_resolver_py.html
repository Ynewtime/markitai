1"""Resolver layer: structured extractor orchestration above root selection. 

2 

3The resolver layer sits between site-specific extractor lookup and the generic 

4pipeline extraction. It provides a typed contract (``ResolvedPage``) that 

5an extractor's optional ``resolve()`` method may return. 

6 

7Design rules: 

8- ``ResolvedPage`` lives here, NOT in ``FetchResult`` or fetch-layer types. 

9- A resolver may return ``content_root`` (a Tag) or ``content_html`` (a string). 

10 It must NEVER return final Markdown. 

11- If no extractor implements ``resolve()``, or no extractor matches the URL, 

12 ``resolve_page()`` returns ``None`` and the generic pipeline remains the 

13 fallback. 

14- Metadata overrides from ``ResolvedPage.metadata_overrides`` are merged over 

15 the generic metadata extracted by ``extract_metadata()``. 

16""" 

17

33@dataclass(slots=True) 

34class ResolvedPage: 

35 """Output of a site-specific resolver. 

36 

37 A resolver may populate either ``content_root`` (a parsed Tag) or 

38 ``content_html`` (a raw HTML string). It must NOT set final Markdown 

39 in either field — the pipeline is responsible for the HTML→Markdown 

40 conversion step. 

41 

42 Attributes: 

43 content_root: Parsed BeautifulSoup Tag representing the content root. 

44 Mutually exclusive with ``content_html`` (caller may set either). 

45 content_html: Raw HTML string of the primary content. 

46 Mutually exclusive with ``content_root``. 

47 metadata_overrides: Mapping of metadata fields to override (e.g. 

48 ``{"title": "Custom Title"}``). Merged over generic metadata. 

49 semantic: Optional semantic models derived during resolution (e.g. 

50 a conversation thread). 

51 diagnostics: Internal debug information populated during resolution. 

52 """ 

53 

54 content_root: Tag | None = None 

55 content_html: str | None = None 

56 metadata_overrides: dict[str, Any] = field(default_factory=dict) 

57 semantic: SemanticExtraction | None = None 

58 diagnostics: dict[str, Any] = field(default_factory=dict)

61def resolve_page( 

62 html: str, 

63 url: str, 

64 *, 

65 resolver: object | None = None, 

66) -> ResolvedPage | None: 

67 """Attempt structured resolution of a page via a site-specific extractor. 

68 

69 Parses ``html``, finds a matching extractor (or uses the provided 

70 ``resolver``), and calls its ``resolve()`` method if present. The result 

71 is validated to ensure it is a ``ResolvedPage`` and not raw Markdown. 

72 

73 Args: 

74 html: Raw HTML source of the page. 

75 url: Canonical URL of the page (used for extractor lookup). 

76 resolver: Optional extractor override. When supplied, skips the 

77 extractor registry lookup and uses this object directly. 

78 

79 Returns: 

80 A ``ResolvedPage`` if a resolver matched and returned structured 

81 content, or ``None`` when falling back to the generic pipeline. 

82 

83 Raises: 

84 TypeError: If the resolver's ``resolve()`` method returns something 

85 other than a ``ResolvedPage`` (e.g. a Markdown string). 

86 """ 

87 extractor = resolver 

88 if extractor is None: 

89 from markitai.webextract.extractors.registry import find_extractor 

90 

91 extractor = find_extractor(url) 

92 

93 if extractor is None: 

94 return None 

95 

96 resolve_fn = getattr(extractor, "resolve", None) 

97 if resolve_fn is None or not callable(resolve_fn): 

98 return None 

99 

100 soup = parse_html(html) 

101 raw_result = resolve_fn(soup, url) 

102 

103 # Validate: resolver must not return final Markdown 

104 _validate_resolver_result(raw_result, extractor) 

105 

106 return raw_result # type: ignore[return-value]

109async def resolve_page_async( 

110 html: str, 

111 url: str, 

112 *, 

113 resolver: object | None = None, 

114 policy: EnrichmentPolicy | None = None, 

115) -> ResolvedPage | None: 

116 """Async variant of ``resolve_page`` that may run policy-aware enrichers. 

117 

118 First runs the sync resolver to obtain a baseline ``ResolvedPage``. 

119 Then, if the policy permits, discovers and runs any applicable enrichers. 

120 If an enricher returns an improved result, that result is used instead. 

121 Enricher failures never propagate — the sync baseline is the fallback. 

122 

123 .. note:: 

124 

125 **Not yet wired into the production extraction path.** The main 

126 ``extract_web_content()`` function currently calls the sync 

127 ``resolve_page()`` only. This async variant and its enricher 

128 infrastructure are available for explicit use but are not 

129 automatically invoked during URL conversion. 

130 

131 Args: 

132 html: Raw HTML source of the page. 

133 url: Canonical URL of the page. 

134 resolver: Optional extractor override (forwarded to ``resolve_page``). 

135 policy: Enrichment policy controlling whether enrichers may run. 

136 Defaults to a permissive policy when ``None``. 

137 

138 Returns: 

139 The best available ``ResolvedPage``, or ``None`` if neither the sync 

140 resolver nor any enricher produced a result. 

141 """ 

142 from markitai.webextract.enrichers import get_enrichers_for_url 

143 from markitai.webextract.enrichers.base import EnrichmentPolicy as _Policy 

144 

145 if policy is None: 

146 policy = _Policy() 

147 

148 # Step 1: run the sync resolver as the baseline 

149 sync_result = resolve_page(html, url, resolver=resolver) 

150 

151 # Step 2: skip enrichers if the policy forbids async or network 

152 if not policy.allow_async or not policy.allow_network: 

153 return sync_result 

154 

155 # Step 3: find enrichers applicable to this URL 

156 candidates = get_enrichers_for_url(url) 

157 if not candidates: 

158 return sync_result 

159 

160 # Step 4: run each enricher; use the first one that returns a result 

161 for enricher in candidates: 

162 if not enricher.should_run(url, policy): 162 ↛ 163line 162 didn't jump to line 163 because the condition on line 162 was never true

163 continue 

164 try: 

165 enriched = await enricher.enrich(url, sync_result) 

166 except Exception as exc: 

167 logger.debug( 

168 "[Resolver] enricher '{}' raised unexpectedly: {}", 

169 enricher.name, 

170 exc, 

171 ) 

172 enriched = None 

173 

174 if enriched is not None: 

175 logger.debug("[Resolver] enricher '{}' improved result", enricher.name) 

176 return enriched 

177 

178 return sync_result
