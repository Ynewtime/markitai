1"""Degeneration guard for VLM outputs. 

2 

3Vision-model extraction can degenerate into repetition loops where the model 

4emits the same line or chunk endlessly at the end of the output. The logits 

5are not reachable through litellm (no ``no_repeat_ngram`` processors), so this 

6module post-checks generated text: detect a degenerate tail, truncate it 

7(keeping one instance of the repeated unit), and let callers skip persisting 

8the salvaged response to cache so a clean retry is possible on the next run. 

9""" 

10 

11from __future__ import annotations 

12 

13from loguru import logger 

14 

15# Minimum length (chars) of a repeated chunk for the chunk detector. 

16DEFAULT_MIN_NGRAM_CHARS = 20 

17# Minimum consecutive repeats of a chunk to flag degeneration. 

18DEFAULT_MIN_REPEATS = 4 

19# Minimum consecutive identical non-empty lines to flag degeneration. 

20DEFAULT_MIN_LINE_REPEATS = 6 

21 

22# Degeneration is a tail property; cap the scanned region for very large texts. 

23_MAX_SCAN_CHARS = 200_000 

24 

25 

26def _detect_repeated_lines(text: str, min_line_repeats: int) -> int | None: 

27 """Detect the same non-empty line repeated at the tail of the text. 

28 

29 Blank lines between repeats are ignored, so both ``line\\nline`` and 

30 ``line\\n\\nline`` loops are caught. Lines that differ in content (e.g. 

31 markdown table rows or list items) never match. 

32 

33 Returns: 

34 Offset just after the first line of the trailing run (so slicing at 

35 the offset keeps one instance), or None if no run is found. 

36 """ 

37 lines = text.splitlines(keepends=True) 

38 starts: list[int] = [] 

39 pos = 0 

40 for line in lines: 

41 starts.append(pos) 

42 pos += len(line) 

43 

44 unit: str | None = None 

45 count = 0 

46 first_idx = -1 

47 for i in range(len(lines) - 1, -1, -1): 

48 stripped = lines[i].strip() 

49 if not stripped: 

50 continue 

51 if unit is None: 

52 unit = stripped 

53 count = 1 

54 first_idx = i 

55 elif stripped == unit: 

56 count += 1 

57 first_idx = i 

58 else: 

59 break 

60 

61 if unit is None or count < min_line_repeats: 

62 return None 

63 return starts[first_idx] + len(lines[first_idx])

66def _detect_repeated_chunk( 

67 text: str, min_ngram_chars: int, min_repeats: int 

68) -> int | None: 

69 """Detect a chunk of >= min_ngram_chars repeated at the tail of the text. 

70 

71 Finds the longest periodic suffix via the KMP prefix function of the 

72 reversed tail (O(n)); a trailing partial repeat (generation cut mid-unit) 

73 is covered by the periodicity check. 

74 

75 Returns: 

76 Offset just after the first full unit of the periodic suffix (so 

77 slicing at the offset keeps one instance), or None. 

78 """ 

79 tail = text[-_MAX_SCAN_CHARS:] 

80 base = len(text) - len(tail) 

81 min_len = min_ngram_chars * min_repeats 

82 n = len(tail) 

83 if n < min_len: 

84 return None 

85 

86 rev = tail[::-1] 

87 pi = [0] * n 

88 k = 0 

89 for i in range(1, n): 

90 while k and rev[i] != rev[k]: 

91 k = pi[k - 1] 

92 if rev[i] == rev[k]: 

93 k += 1 

94 pi[i] = k 

95 

96 # Prefix of rev of length L == suffix of text of length L; smallest period 

97 # is L - pi[L-1]. Scan longest-first so the full degenerate run is found. 

98 for i in range(n - 1, min_len - 2, -1): 

99 length = i + 1 

100 period = length - pi[i] 

101 if period >= min_ngram_chars and period * min_repeats <= length: 

102 return base + n - length + period 

103 return None

106def detect_trailing_repetition( 

107 text: str, 

108 min_ngram_chars: int = DEFAULT_MIN_NGRAM_CHARS, 

109 min_repeats: int = DEFAULT_MIN_REPEATS, 

110 min_line_repeats: int = DEFAULT_MIN_LINE_REPEATS, 

111) -> int | None: 

112 """Detect degenerate repetition at the tail of a text. 

113 

114 Combines two detectors: 

115 - Repeated line: the same non-empty line >= min_line_repeats times 

116 consecutively at the tail (catches short-unit loops). Preferred when 

117 it fires because its cut lands exactly on a line boundary. 

118 - Repeated chunk: a unit of >= min_ngram_chars chars repeated 

119 >= min_repeats times consecutively at the tail (catches long and 

120 multi-line loops, including a partial final repeat). Its cut may land 

121 on a rotation of the semantic unit. 

122 

123 Legitimate repetition (markdown table rows or list items with differing 

124 content) is not flagged because the repeated units are not identical. 

125 

126 Args: 

127 text: Text to inspect (typically VLM output). 

128 min_ngram_chars: Minimum repeated-chunk length in characters. 

129 min_repeats: Minimum consecutive chunk repeats. 

130 min_line_repeats: Minimum consecutive identical non-empty lines. 

131 

132 Returns: 

133 Offset where the redundant tail starts (text[:offset] keeps exactly 

134 one instance of the repeated unit), or None if no degeneration. 

135 """ 

136 if not text: 

137 return None 

138 line_offset = _detect_repeated_lines(text, min_line_repeats) 

139 if line_offset is not None: 

140 return line_offset 

141 return _detect_repeated_chunk(text, min_ngram_chars, min_repeats)

144def truncate_degenerate_tail( 

145 text: str, 

146 *, 

147 context: str = "", 

148 stage: str = "", 

149) -> tuple[str, bool]: 

150 """Detect and truncate a degenerate repetition tail in VLM output. 

151 

152 Returns: 

153 Tuple of (text, truncated). When truncated is True the redundant tail 

154 was removed (one instance of the repeated unit is kept) and the caller 

155 should skip persisting the response to cache so a clean retry on the 

156 next run is not poisoned. 

157 """ 

158 if not text: 

159 return text, False 

160 offset = detect_trailing_repetition(text) 

161 if offset is None or offset <= 0 or offset >= len(text): 

162 return text, False 

163 logger.warning( 

164 "[{}] {}: VLM output degenerated into trailing repetition; " 

165 "truncating {} of {} chars (keeping one instance of the repeated unit)", 

166 context or "unknown", 

167 stage or "vlm", 

168 len(text) - offset, 

169 len(text), 

170 ) 

171 return text[:offset].rstrip(), True
