1"""Fetch policy engine for determining URL fetch strategies. 

2 

3Strategy priority rationale (v0.15.0, local-first): 

4 static → playwright → defuddle → jina → cloudflare 

5 

6- static: Fast local fetch + native webextract pipeline (full defuddle port: 

7 scoring, math, footnotes, code protection) — matches remote defuddle 

8 quality on the ground-truth corpus, no data leaves the machine. 

9- playwright: Full JS rendering for SPA/JS-heavy pages, local. Includes 

10 oEmbed enricher fallback for X/Twitter URLs when DOM parsing fails. 

11- defuddle: Remote extraction API (free, no auth). Consent-gated 

12 (fetch.remote_consent) since it receives the user's URLs. 

13- jina: Remote Reader API, free tier (20 RPM), consent-gated; anonymous 

14 access is blocked for some domains (e.g. github.com → 451). 

15- cloudflare: Requires CF account credentials, rate-limited, consent-gated. 

16"""

29def _extract_host(domain: str) -> str: 

30 """Extract host from a netloc-like domain string.""" 

31 if domain.startswith("[") and "]" in domain: 

32 return domain[1 : domain.index("]")] 

33 # Bare IPv6 (multiple colons, e.g. "fd12::1") — return as-is 

34 if domain.count(":") > 1: 

35 return domain 

36 return domain.split(":", 1)[0]

76def match_local_only(domain: str, patterns: list[str]) -> bool: 

77 """Check if domain matches any local-only exemption pattern. 

78 

79 Supports NO_PROXY syntax: 

80 - Domain exact: ``internal.corp.com`` 

81 - Suffix: ``.internal.com`` (matches subdomains only) 

82 - Wildcard: ``*.internal.com`` (same as ``.internal.com``) 

83 - IP exact: ``10.0.1.5`` 

84 - CIDR: ``10.0.0.0/8``, ``fd00::/8`` 

85 - Star: ``*`` (matches everything) 

86 - Special: ``localhost`` 

87 """ 

88 if not patterns: 

89 return False 

90 

91 host = _extract_host(domain).strip().lower() 

92 if not host: 92 ↛ 93line 92 didn't jump to line 93 because the condition on line 92 was never true

93 return False 

94 

95 for pattern in patterns: 

96 p = pattern.strip().lower() 

97 if not p: 97 ↛ 98line 97 didn't jump to line 98 because the condition on line 97 was never true

98 continue 

99 

100 # Wildcard: match everything 

101 if p == "*": 

102 return True 

103 

104 # Normalize *.foo.com → .foo.com 

105 if p.startswith("*."): 

106 p = p[1:] # "*.foo.com" → ".foo.com" 

107 

108 # Suffix match: .foo.com matches sub.foo.com but not foo.com 

109 if p.startswith("."): 

110 if host.endswith(p): 

111 return True 

112 continue 

113 

114 # CIDR match 

115 if "/" in p: 

116 try: 

117 network = ipaddress.ip_network(p, strict=False) 

118 host_ip = ipaddress.ip_address(host) 

119 if host_ip in network: 

120 return True 

121 except ValueError: 

122 pass 

123 continue 

124 

125 # Exact match (domain or IP) 

126 if host == p: 

127 return True 

128 

129 return False

143 def decide( 

144 self, 

145 domain: str, 

146 known_spa: bool, 

147 explicit_strategy: str | None, 

148 fallback_patterns: list[str], 

149 policy_enabled: bool, 

150 has_jina_key: bool = False, 

151 domain_prefer_strategy: str | None = None, 

152 global_strategy_priority: list[str] | None = None, 

153 domain_strategy_priority: list[str] | None = None, 

154 local_only_patterns: list[str] | None = None, 

155 ) -> FetchDecision: 

156 """Decide the fetch strategy order. 

157 

158 Priority chain (highest to lowest): 

159 1. Explicit strategy (CLI flag) -- single strategy, no fallback 

160 2. Local-only exemption -- security: restrict to local strategies 

161 3. Private/local domain -- hardcoded local detection 

162 4. Domain strategy_priority -- per-domain full override 

163 5. Domain prefer_strategy -- per-domain single promotion 

164 6. Global strategy_priority -- global full override 

165 7. SPA/JS-required pattern -- browser-first order 

166 8. Default order 

167 """ 

168 # 1. Explicit strategy always wins 

169 if explicit_strategy and explicit_strategy != "auto": 

170 return FetchDecision( 

171 order=[explicit_strategy], reason=f"explicit_{explicit_strategy}" 

172 ) 

173 

174 is_fallback_domain = any( 

175 domain == p or domain.endswith(f".{p}") for p in fallback_patterns 

176 ) 

177 

178 # 2. Local-only exemption (security: skip external APIs) 

179 if local_only_patterns and match_local_only(domain, local_only_patterns): 

180 order = ( 

181 ["playwright", "static"] 

182 if known_spa or is_fallback_domain 

183 else LOCAL_ONLY_STRATEGIES.copy() 

184 ) 

185 return FetchDecision(order=order, reason="local_only_pattern") 

186 

187 # 3. Private/local domain (hardcoded detection) 

188 if is_private_or_local_domain(domain): 

189 order = ( 

190 ["playwright", "static"] 

191 if known_spa or is_fallback_domain 

192 else LOCAL_ONLY_STRATEGIES.copy() 

193 ) 

194 return FetchDecision(order=order, reason="private_or_local") 

195 

196 # 4. Per-domain strategy_priority (full override) 

197 if domain_strategy_priority: 

198 return FetchDecision( 

199 order=list(domain_strategy_priority), reason="domain_priority" 

200 ) 

201 

202 # 5. Per-domain prefer_strategy (single promotion) 

203 if domain_prefer_strategy: 

204 remaining = [s for s in ALL_STRATEGIES if s != domain_prefer_strategy] 

205 return FetchDecision( 

206 order=[domain_prefer_strategy] + remaining, 

207 reason=f"domain_prefer_{domain_prefer_strategy}", 

208 ) 

209 

210 # 6. Global strategy_priority (full override) 

211 if global_strategy_priority: 

212 return FetchDecision( 

213 order=list(global_strategy_priority), reason="global_priority" 

214 ) 

215 

216 if not policy_enabled: 

217 return FetchDecision( 

218 order=ALL_STRATEGIES.copy(), 

219 reason="disabled", 

220 ) 

221 

222 # 7. SPA/JS-heavy: browser first 

223 # Known SPAs and fallback-pattern domains need JS rendering — go 

224 # straight to the local browser; consent-gated remote services are 

225 # the fallback, and static (which already failed to produce content 

226 # for such domains) goes last. 

227 if known_spa or is_fallback_domain: 

228 return FetchDecision( 

229 order=["playwright", "defuddle", "jina", "cloudflare", "static"], 

230 reason="spa_or_pattern", 

231 ) 

232 

233 # 8. Default 

234 return FetchDecision( 

235 order=ALL_STRATEGIES.copy(), 

236 reason="default", 

237 )
